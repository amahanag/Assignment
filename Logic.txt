package logic;

import java.util.Scanner;

public class logic {

	public static void main(String[] args) {
	Scanner sc=new Scanner(System.in);
	int a=sc.nextInt();
	int b=sc.nextInt();
	swap(a,b);
	swap1(a,b);
	
	}
	
	public static void swap(int a,int b)
	{
		int temp;
		temp=a;
		a=b;
		b=temp;
		
		System.out.println("By third variable"+a+" "+b);
	}
	
	public static void swap1(int a,int b)
	{
		a=a^b;
		b=a^b;
		a=a^b;
		System.out.println("By without third variable"+a+" "+b);
	}
	

}
